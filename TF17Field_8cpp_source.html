184 Vector3d TF17Field::getDiskField(const double& r, const double& z, const double& phi, const double& sinPhi, const double& cosPhi) const {
185  Vector3d b(0.);
186  double B_r = 0;
187  double B_phi = 0;
188  double B_z = 0;
189 
190  if (disk_model == TF17DiskModel::Ad1) { // ==========================================================
191  if (r > r1_disk) {
192  double z1_disk_z = (1. + a_disk * r1_disk * r1_disk) / (1. + a_disk * r * r); // z1_disk / z
193  // B components in (r, phi, z)
194  double B_r0 = radialFieldScale(B1_disk, phi_star_disk, z1_disk_z*z, phi, r, z);
195  B_r = (r1_disk / r) * z1_disk_z * B_r0;
196  B_z = 2 * a_disk * r1_disk * z1_disk_z*z / (1+ a_disk * r * r) * B_r0;
197  B_phi = azimuthalFieldComponent(r, z, B_r, B_z);
198  } else {
199  // within r = r1_disk, the field lines are straight in direction g_phi + phi_star_disk
200  // and thus z = z1
201  double phi1_disk = shiftedWindingFunction(r1_disk, z) + phi_star_disk;
202  double B_amp = B1_disk * exp(-fabs(z) / H_disk);
203  B_r = cos(phi1_disk - phi) * B_amp;
204  B_phi = sin(phi1_disk - phi) * B_amp;
205  }
206 
207  } else if (disk_model == TF17DiskModel::Bd1) { // ===================================================
208  // for model Bd1, best fit for n = 2
209  if ( r > epsilon ) {
210  double r1_disk_r = r1_disk / r;
211  double z1_disk_z = 5. / (r1_disk_r*r1_disk_r + 4./sqrt(r1_disk_r)); // z1_disk / z -> remove z dependancy
212  double B_r0 = radialFieldScale(B1_disk, phi_star_disk, z1_disk_z*z, phi, r, z);
213  B_r = r1_disk_r * z1_disk_z * B_r0;
214  B_z = -0.4 * r1_disk_r / r * z1_disk_z* z1_disk_z * z * (r1_disk_r*r1_disk_r - 1./sqrt(r1_disk_r)) * B_r0;
215  } else {
216  double z1_disk_z = 5. * r*r / (r1_disk*r1_disk); // z1_disk / z -> remove z dependancy
217  double B_r0 = radialFieldScale(B1_disk, phi_star_disk, z1_disk_z*z, phi, r, z);
218  B_r = 5. * r / r1_disk * B_r0;
219  B_z = -10. * z / r1_disk * B_r0;
220  }
221  B_phi = azimuthalFieldComponent(r, z, B_r, B_z);
222 
223  } else if (disk_model == TF17DiskModel::Dd1) { // ===================================================
224  // for model Dd1, best fit for n = 0.5
225  double z_sign = z >= 0 ? 1. : -1.;
226  double z_abs = fabs(z);
227  if ( z_abs > epsilon ) {
228  double z1_disk_z = z1_disk / z_abs;
229  double r1_disk_r = 1.5 / (sqrt(z1_disk_z) + 0.5/z1_disk_z); // r1_disk / r
230  double F_r = r1_disk_r*r <= L_disk ? 1. : exp(1. - r1_disk_r*r/L_disk);
231  // simplication of the equation in the cosinus
232  double B_z0 = z_sign * B1_disk * F_r * cos(phi - shiftedWindingFunction(r, z) - phi_star_disk);
233  B_r = -0.5/1.5 * r1_disk_r * r1_disk_r * r1_disk_r * r / z_abs * (sqrt(z1_disk_z) - 1/z1_disk_z) * B_z0;
234  B_z = z_sign * r1_disk_r * r1_disk_r * B_z0;
235  } else {
236  double z_z1_disk = z_abs / z1_disk;
237  double r1_disk_r = 1.5 * sqrt(z_abs / z1_disk); // r1_disk / r
238  double F_r = r1_disk_r*r <= L_disk ? 1. : exp(1. - r1_disk_r*r/L_disk);
239  double B_z0 = z_sign * B1_disk * F_r * cos(phi - shiftedWindingFunction(r, z) - phi_star_disk);
240  B_r = -1.125 * r / z1_disk * (1 - 2.5 * z_z1_disk * sqrt(z_z1_disk)) * B_z0;
241  B_z = z_sign * r1_disk_r * r1_disk_r * B_z0;
242  }
243  B_phi = azimuthalFieldComponent(r, z, B_r, B_z);
244  }
245 
246  // Convert to (x, y, z) components
247  b.x = - (B_r * cosPhi - B_phi * sinPhi); // flip x-component at the end
248  b.y = B_r * sinPhi + B_phi * cosPhi;
249  b.z = B_z;
250  return b;
251 }

253 Vector3d TF17Field::getHaloField(const double& r, const double& z, const double& phi, const double& sinPhi, const double& cosPhi) const {
254  int m;
255 
256  Vector3d b(0.);
257  double r1_halo_r = (1. + a_halo * z1_halo * z1_halo) / (1. + a_halo * z * z);
258  // B components in (r, phi, z)
259  double B_z0;
260 
261  if (halo_model == TF17HaloModel::C0) { // m = 0
262  B_z0 = B1_halo * exp(-r1_halo_r*r / L_halo);
263  } else if (halo_model == TF17HaloModel::C1) { // m = 1
264  // simplication of the equation in the cosinus
265  double phi_prime = phi - shiftedWindingFunction(r, z) - phi_star_halo;
266  B_z0 = B1_halo * exp(-r1_halo_r*r / L_halo) * cos(phi_prime);
267  }
268 
269  // Contrary to article, Br has been rewriten to a little bit by replacing
270  // (2 * a * r1**3 * z) / (r**2) by (2 * a * r1**2 * z) / (r * (1+a*z**2))
271  // but that is strictly equivalent except we can reintroduce the z1 in the expression via r1
272  double B_r = 2 * a_halo * r1_halo_r * r1_halo_r * r * z / (1. + a_halo * z * z) * B_z0;
273  double B_z = r1_halo_r * r1_halo_r * B_z0;
274  double B_phi = azimuthalFieldComponent(r, z, B_r, B_z);
275 
276  // Convert to (x, y, z) components
277  b.x = - (B_r * cosPhi - B_phi * sinPhi); // flip x-component at the end
278  b.y = B_r * sinPhi + B_phi * cosPhi;
279  b.z = B_z;
280 
281  return b;
282 }

284 double TF17Field::azimuthalFieldComponent(const double& r, const double& z, const double& B_r, const double& B_z) const {
285  double r_ = r / L_p;
286  double rscale = r > epsilon ? r_ * exp(-r_) / (1 - exp(-r_)) : 1 - r_/2. - r_*r_/12. ;
287  double B_phi = cot_p0 / zscale(z) * rscale * B_r;
288  B_phi = B_phi - 2 * z * r / (H_p * H_p) / zscale(z) * shiftedWindingFunction(r, z) * B_z;
289  return B_phi;
290 }
