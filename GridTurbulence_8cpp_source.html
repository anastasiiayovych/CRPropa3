176  Bky = (fftwf_complex*) fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
177  Bkz = (fftwf_complex*) fftwf_malloc(sizeof(fftwf_complex) * n * n * n2);
178

180  if (seed != 0)
181  random.seed(seed); // use given seed
182 
183  // calculate the n possible discrete wave numbers
184  double K[n];
185  for (int i = 0; i < n; i++)
186  K[i] = (double) i / n - i / (n / 2);
187 
188  // construct the field in configuration space
189  int i;
190  double k;
191 
192  // parameters goes for non-helical calculations
193  double theta, phase, cosPhase, sinPhase;
194 
195  double kMin = spacing.x / lMax;
196  double kMax = spacing.x / lMin;
197  Vector3f b; // real b-field vector
198  Vector3f ek, e1, e2; // orthogonal base
199  Vector3f n0(1, 1, 1); // arbitrary vector to construct orthogonal base
200 
201  for (size_t ix = 0; ix < n; ix++) {
202  for (size_t iy = 0; iy < n; iy++) {
203  for (size_t iz = 0; iz < n2; iz++) {
204 
205  i = ix * n * n2 + iy * n2 + iz;
206  ek.setXYZ(K[ix], K[iy], K[iz]);
207  k = ek.getR();
208 
209  // wave outside of turbulent range -> B(k) = 0
210  if ((k < kMin) || (k > kMax)) {
211  Bkx[i][0] = 0;
212  Bkx[i][1] = 0;
213  Bky[i][0] = 0;
214  Bky[i][1] = 0;
215  Bkz[i][0] = 0;
216  Bkz[i][1] = 0;
217  continue;
218  }
219 
220  // construct an orthogonal base ek, e1, e2
221  if (ek.isParallelTo(n0, float(1e-3))) {
222  // ek parallel to (1,1,1)
223  e1.setXYZ(-1., 1., 0);
224  e2.setXYZ(1., 1., -2.);
225  } else {
226  // ek not parallel to (1,1,1)
227  e1 = n0.cross(ek);
228  e2 = ek.cross(e1);
229  }
230  e1 /= e1.getR();
231  e2 /= e2.getR();
232 
233  // random orientation perpendicular to k
234  theta = 2 * M_PI * random.rand();
235  b = e1 * cos(theta) + e2 * sin(theta);
236 
237  // normal distributed amplitude with mean = 0 and sigma = k^alpha/2
238  b *= random.randNorm() * pow(k, alpha / 2);
239 
240  // uniform random phase
241  phase = 2 * M_PI * random.rand();
242  cosPhase = cos(phase); // real part
243  sinPhase = sin(phase); // imaginary part
244 
245  Bkx[i][0] = b.x * cosPhase;
246  Bkx[i][1] = b.x * sinPhase;
247  Bky[i][0] = b.y * cosPhase;
248  Bky[i][1] = b.y * sinPhase;
249  Bkz[i][0] = b.z * cosPhase;
250  Bkz[i][1] = b.z * sinPhase;
251  } // for iz
252  } // for iy
253  } // for ix
254 
255  executeInverseFFTInplace(grid, Bkx, Bky, Bkz);
256 
257  fftwf_free(Bkx);
258  fftwf_free(Bky);
259  fftwf_free(Bkz);
260
