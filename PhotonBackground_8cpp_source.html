138 double PhotonFieldSampling::sample_eps(bool onProton, double E_in, double z_in) const {
139  if (bgFlag == 0)
140  throw std::runtime_error("error: select photon field first: 1 (CMB) or 2 (IRB_Kneiske04)");
141 
142  const double mass = onProton? 0.93827 : 0.93947; // Gev/c^2
143  const double P_in = sqrt(E_in * E_in - mass * mass); // GeV/c
144 
145  double eps = 0.;
146  if (bgFlag == 1) {
147  // CMB
148  const double tbb = 2.73 * (1. + z_in);
149  const double epsMin = (1.1646 - mass * mass) / 2. / (E_in + P_in) * 1.e9;
150  const double epsMax = 0.007 * tbb;
151  if (epsMin > epsMax) {
152  std::cout << "sample_eps (CMB): CMF energy is below threshold for nucleon energy " << E_in << " GeV !" << std::endl;
153  return 0.;
154  }
155 
156  const double cnorm = gaussInt([this, onProton, E_in, z_in](double e) { return this->prob_eps(e, onProton, E_in, z_in); }, epsMin, epsMax);
157  const double epskt = 8.619e-5 * tbb;
158  const double epspmax = (3.e-3 * std::pow(E_in * epskt * 1.e-9, -0.97) + 0.047) / 3.9e2 * tbb;
159  const double pmaxc = prob_eps(epspmax, onProton, E_in, z_in) / cnorm;
160  const double facpmax = 1.6;
161  const double pMax = facpmax * pmaxc;
162 
163  // sample eps between epsMin ... epsMax
164  double peps = 0.;
165  Random &random = Random::instance();
166  do {
167  eps = epsMin + random.rand() * (epsMax - epsMin);
168  peps = prob_eps(eps, onProton, E_in, z_in) / cnorm;
169  } while (random.rand() * pMax > peps);
170  }
171 
172  if (bgFlag == 2) {
173  // IRB_Kneiske04
174  const double epsMin = std::max(0.00395, 1.e9 * (1.1646 - mass * mass) / 2. / (E_in + P_in)); // eV
175  const double epsMax = 12.2; // eV
176  if (epsMin > epsMax) {
177  std::cout << "sample_eps (IRB): CMF energy is below threshold for nucleon energy " << E_in << " GeV !" << std::endl;
178  return 0.;
179  }
180  const int i_max = static_cast<int>(10. * std::log(epsMax / epsMin)) + 1;
181  const double de = std::log(epsMax / epsMin) / i_max;
182  double rmax = 0.;
183  double eps_dum = 0.;
184  double dum = 0.;
185  for (int i = 0; i < i_max; ++i) {
186  eps_dum = epsMin * std::exp(i * de);
187  dum = eps_dum * eps_dum * getPhotonDensity(eps_dum, z_in);
188  if (dum > rmax)
189  rmax = dum;
190  }
191  const double beta = 4.;
192  const double e1 = std::pow(epsMin, 1. - beta);
193  const double e2 = std::pow(epsMax, 1. - beta);
194  bool keepTrying = true;
195  int i_rep = 0;
196  Random &random = Random::instance();
197  do {
198  if (i_rep >= 100000) {
199  keepTrying = false;
200  return 0.;
201  }
202  eps = std::pow(random.rand() * (e1 - e2) + e2, 1./(1. - beta));
203  i_rep++;
204  if (random.rand() < eps * eps * getPhotonDensity(eps, z_in) / rmax)
205  keepTrying = false;
206  } while (keepTrying);
207  }
208  return eps * eV;
209 }

211 double PhotonFieldSampling::prob_eps(double eps, bool onProton, double E_in, double z_in) const {
212  const double mass = onProton? 0.93827 : 0.93947; // Gev/c^2
213  double gamma = E_in / mass;
214  double beta = std::sqrt(1. - 1. / gamma / gamma);
215  double photonDensity = getPhotonDensity(eps, z_in);
216  if (photonDensity == 0.) {
217  return 0.;
218  } else {
219  double sMin = 1.1646; // [GeV], head-on collision
220  double sMax = std::max(sMin, mass * mass + 2. * eps / 1.e9 * E_in * (1. + beta));
221  double sintegr = gaussInt([this, onProton](double s) { return this->functs(s, onProton); }, sMin, sMax);
222  return photonDensity / eps / eps * sintegr / 8. / beta / E_in / E_in * 1.e18 * 1.e6;
223  }

226 double PhotonFieldSampling::getPhotonDensity(double eps, double z_in) const {
227  if (bgFlag == 1) {
228  // CMB
229  return 1.318e13 * eps * eps / (std::exp(eps / (8.619e-5 * 2.73)) - 1.);
230  }
231 
232  if (bgFlag == 2) {
233  // IR background from Primack et al. (1999)
234  const double ZMAX_IR = 5.;
235  if (z_in > ZMAX_IR)
236  return 0.;
237  const double X = 1.2398 * (1. + z_in) / eps;
238  if (X > 500.)
239  return 0.;
240 
241  static const double XData[15] = {-1.0, -0.75, -0.5, -0.25, 0.0, 0.25, 0.5, 0.75, 1.0, 1.25, 1.5, 1.75, 2.0, 2.25, 2.5}; // log10(eV)
242  static const double YData[15] = {-0.214401, 0.349313, 0.720354, 0.890389, 1.16042, 1.24692, 1.06525, 0.668659, 0.536312, 0.595859, 0.457456, 0.623521, 1.20208, 1.33657, 1.04461}; // log10(nW/m^2/sr)
243 
244  // interpolate
245  if (std::log10(X) <= XData[0])
246  return 0.;
247  if (std::log10(X) >= XData[14]) {
248  double result = (YData[14] - YData[13]) / (XData[14] - XData[13]) * (std::log10(X) - XData[13]) + YData[13];
249  return std::pow(10., result);
250  }
251  int index = 1;
252  do {
253  index++;
254  } while (XData[index] < std::log10(X));
255  double result = (YData[index] - YData[index - 1]) / (XData[index] - XData[index - 1]) * (std::log10(X) - XData[index - 1]) + YData[index - 1];
256  result = std::pow(10., result);
257 
258  const double fluxConversion = 3.82182e3; // conversion from nW/cm^3/sr to eV/cm^3
259  return result * std::pow(1. + z_in, 4.) / (eps * eps) / fluxConversion;
260  }

263 double PhotonFieldSampling::crossection(double x, bool onProton) const {
264  const double mass = onProton? 0.93827 : 0.93947; // Gev/c^2
265  const double sth = 1.1646; // GeV^2
266  const double s = mass * mass + 2. * mass * x;
267  if (s < sth)
268  return 0.;
269  double cross_res = 0.;
270  double cross_dir = 0.;
271  double cross_dir1 = 0.;
272  double cross_dir2 = 0.;
273  double sig_res[9];
274 
275  // first half of array: 9x proton resonance data | second half of array 9x neutron resonance data
276  static const double AMRES[18] = {1.231, 1.440, 1.515, 1.525, 1.675, 1.680, 1.690, 1.895, 1.950, 1.231, 1.440, 1.515, 1.525, 1.675, 1.675, 1.690, 1.895, 1.950};
277  static const double BGAMMA[18] = {5.6, 0.5, 4.6, 2.5, 1.0, 2.1, 2.0, 0.2, 1.0, 6.1, 0.3, 4.0, 2.5, 0.0, 0.2, 2.0, 0.2, 1.0};
278  static const double WIDTH[18] = {0.11, 0.35, 0.11, 0.1, 0.16, 0.125, 0.29, 0.35, 0.3, 0.11, 0.35, 0.11, 0.1, 0.16, 0.150, 0.29, 0.35, 0.3};
279  static const double RATIOJ[18] = {1., 0.5, 1., 0.5, 0.5, 1.5, 1., 1.5, 2., 1., 0.5, 1., 0.5, 0.5, 1.5, 1., 1.5, 2.};
280  static const double AM2[2] = {0.882792, 0.880351};
281 
282  const int idx = onProton? 0 : 9;
283  double SIG0[9];
284  for (int i = 0; i < 9; ++i) {
285  SIG0[i] = 4.893089117 / AM2[int(onProton)] * RATIOJ[i + idx] * BGAMMA[i + idx];
286  }
287  if (x <= 10.) {
288  cross_res = breitwigner(SIG0[0], WIDTH[0 + idx], AMRES[0 + idx], x, onProton) * Ef(x, 0.152, 0.17);
289  sig_res[0] = cross_res;
290  for (int i = 1; i < 9; ++i) {
291  sig_res[i] = breitwigner(SIG0[i], WIDTH[i + idx], AMRES[i + idx], x, onProton) * Ef(x, 0.15, 0.38);
292  cross_res += sig_res[i];
293  }
294  // direct channel
295  if ((x > 0.1) && (x < 0.6)) {
296  cross_dir1 = 92.7 * Pl(x, 0.152, 0.25, 2.0) // single pion production
297  + 40. * std::exp(-(x - 0.29) * (x - 0.29) / 0.002)
298  - 15. * std::exp(-(x - 0.37) * (x - 0.37) / 0.002);
299  } else {
300  cross_dir1 = 92.7 * Pl(x, 0.152, 0.25, 2.0); // single pion production
301  }
302  cross_dir2 = 37.7 * Pl(x, 0.4, 0.6, 2.0); // double pion production
303  cross_dir = cross_dir1 + cross_dir2;
304  }
305  // fragmentation 2:
306  double cross_frag2 = onProton? 80.3 : 60.2;
307  cross_frag2 *= Ef(x, 0.5, 0.1) * std::pow(s, -0.34);
308  // multipion production/fragmentation 1 cross section
309  double cs_multidiff = 0.;
310  double cs_multi = 0.;
311  double cross_diffr1 = 0.;
312  double cross_diffr2 = 0.;
313  double cross_diffr = 0.;
314  if (x > 0.85) {
315  double ss1 = (x - 0.85) / 0.69;
316  double ss2 = onProton? 29.3 : 26.4;
317  ss2 *= std::pow(s, -0.34) + 59.3 * std::pow(s, 0.095);
318  cs_multidiff = (1. - std::exp(-ss1)) * ss2;
319  cs_multi = 0.89 * cs_multidiff;
320  // diffractive scattering:
321  cross_diffr1 = 0.099 * cs_multidiff;
322  cross_diffr2 = 0.011 * cs_multidiff;
323  cross_diffr = 0.11 * cs_multidiff;
324  // **************************************
325  ss1 = std::pow(x - 0.85, 0.75) / 0.64;
326  ss2 = 74.1 * std::pow(x, -0.44) + 62. * std::pow(s, 0.08);
327  double cs_tmp = 0.96 * (1. - std::exp(-ss1)) * ss2;
328  cross_diffr1 = 0.14 * cs_tmp;
329  cross_diffr2 = 0.013 * cs_tmp;
330  double cs_delta = cross_frag2 - (cross_diffr1 + cross_diffr2 - cross_diffr);
331  if (cs_delta < 0.) {
332  cross_frag2 = 0.;
333  cs_multi += cs_delta;
334  } else {
335  cross_frag2 = cs_delta;
336  }
337  cross_diffr = cross_diffr1 + cross_diffr2;
338  cs_multidiff = cs_multi + cross_diffr;
339  // in the original SOPHIA code, here is a switch for the return argument.
340  // Here, only one case (compare in SOPHIA: NDIR=3) is needed.
341  }
342  return cross_res + cross_dir + cs_multidiff + cross_frag2;
343 }

367 double PhotonFieldSampling::breitwigner(double sigma_0, double Gamma, double DMM, double epsPrime, bool onProton) const {
368  const double mass = onProton? 0.93827 : 0.93947; // Gev/c^2
369  const double s = mass * mass + 2. * mass * epsPrime;
370  const double gam2s = Gamma * Gamma * s;
371  return sigma_0 * (s / epsPrime / epsPrime) * gam2s / ((s - DMM * DMM) * (s - DMM * DMM) + gam2s);
